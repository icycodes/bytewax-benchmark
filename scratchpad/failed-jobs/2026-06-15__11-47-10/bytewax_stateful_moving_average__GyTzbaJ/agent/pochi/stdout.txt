from pathlib import Path
from bytewax.outputs import DynamicSink, StatelessSinkPartition
from bytewax.dataflow import Dataflow
import bytewax.operators as op
from bytewax.testing import TestingSource, run_main

class SimpleFileSinkPartition(StatelessSinkPartition):
    def __init__(self, file_path):
        self._file = open(file_path, 'w')
    def write_batch(self, items):
        for item in items:
            self._file.write(item + '\n')
            self._file.flush()
    def close(self):
        self._file.close()

class SimpleFileSink(DynamicSink):
    def __init__(self, file_path):
        self._file_path = file_path
    def build(self, step_id, worker_index, worker_count):
        return SimpleFileSinkPartition(self._file_path)

flow = Dataflow('test_write')
stream = op.input('inp', flow, TestingSource(['hello', 'world']))
