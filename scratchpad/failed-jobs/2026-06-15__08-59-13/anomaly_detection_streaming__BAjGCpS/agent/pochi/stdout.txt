from datetime import datetime, timedelta, timezone
from bytewax.dataflow import Dataflow
import bytewax.operators as op
from bytewax.operators.windowing import SlidingWindower, EventClock, collect_window
from bytewax.testing import TestingSource, run_main

flow = Dataflow('test_sliding')
data = [
    ('sensor_1', {'timestamp': '2026-01-01T12:00:00Z', 'value': 20.0}),
    ('sensor_1', {'timestamp': '2026-01-01T12:00:05Z', 'value': 20.0}),
    ('sensor_1', {'timestamp': '2026-01-01T12:00:10Z', 'value': 20.0}),
]
up = op.input('input', flow, TestingSource(data))

def extract_timestamp(x):
    ts_str = x['timestamp']
    if ts_str.endswith('Z'):
        ts_str = ts_str[:-1] + '+00:00'
    return datetime.fromisoformat(ts_str)

clock = EventClock(
    ts_getter=extract_timestamp,
    wait_for_system_duration=timedelta(seconds=0),
)

windower = SlidingWindower(
    length=timedelta(seconds=60),
    offset=timedelta(seconds=20),
    align_to=datetime(2026, 1, 1, 12, 0, tzinfo=timezone.utc)
)

windowed = collect_window('collect', up, clock, windower)
op.inspect('inspect', windowed.down)
run_main(flow)

from datetime import datetime, timedelta, timezone
import json
import math

T0 = datetime(2026, 1, 1, 12, 0, 0, tzinfo=timezone.utc)
length = timedelta(seconds=60)
offset = timedelta(seconds=20)

events = []
with open('/home/user/anomaly_detection/data.jsonl') as f:
    for line in f:
        item = json.loads(line)
        ts_str = item['timestamp']
        if ts_str.endswith('Z'):
            ts_str = ts_str[:-1] + '+00:00'
        dt = datetime.fromisoformat(ts_str)
        events.append((dt, item))

# Find the range of window indices
# Since events start at 12:00:00 and end at 12:02:00, let's look at window indices from -5 to 10
windows = {}
for dt, item in events:
    # A window starts at T0 + i * offset and ends at T0 + i * offset + length
    # We want T0 + i * offset <= dt < T0 + i * offset + length
    # i * offset <= dt - T0 < i * offset + length
    # (dt - T0 - length) / offset < i <= (dt - T0) / offset
    # Let's just find all i that satisfy this
    diff_sec = (dt - T0).total_seconds()
    length_sec = length.total_seconds()
    offset_sec = offset.total_seconds()
    
    min_i = math.ceil((diff_sec - length_sec) / offset_sec)
    max_i = math.floor(diff_sec / offset_sec)
    
    for i in range(min_i, max_i + 1):
        windows.setdefault(i, []).append(item)

for i in sorted(windows.keys()):
    items = windows[i]
    values = [x['value'] for x in items]
    mean = sum(values) / len(values)
    # Let's calculate population std dev
    p_var = sum((v - mean)**2 for x in values) / len(values)
    p_std = math.sqrt(p_var)
    
    # Let's calculate sample std dev
    s_std = 0.0
    if len(values) > 1:
        s_var = sum((v - mean)**2 for v in values) / (len(values) - 1)
        s_std = math.sqrt(s_var)
        
    print(f'Window {i} (starts at {T0 + i * offset}):')
    print(f'  Count: {len(values)}, Values: {values}')
    print(f'  Mean: {mean:.4f}, Pop Std: {p_std:.4f}, Sample Std: {s_std:.4f}')
    
    # Check outliers (pop)
    for item in items:
        val = item['value']
        if p_std > 0:
            is_out = val > mean + 3 * p_std or val < mean - 3 * p_std
            if is_out:
