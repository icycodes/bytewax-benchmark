import json
from bytewax.dataflow import Dataflow
import bytewax.operators as op
from bytewax.testing import TestingSource, run_main
from bytewax.connectors.stdio import StdOutSink

flow = Dataflow('test_flow')
events = [('prod_1', 1), ('prod_2', 1), ('prod_1', 1)]
s = op.input('inp', flow, TestingSource(events))

def update_count(state, val):
    if state is None:
        state = 0
    state += val
    return (state, state)

counts = op.stateful_map('counts', s, update_count)

def rekey_global(item):
    product_id, count = item
    return ('global', (product_id, count))

global_stream = op.map('rekey', counts, rekey_global)

def update_global_top_3(state, value):
    if state is None:
        product_counts = {}
        last_top_3 = []
    else:
        product_counts = dict(state[0])
        last_top_3 = list(state[1])
    
    product_id, count = value
    product_counts[product_id] = count
    
    sorted_products = sorted(product_counts.items(), key=lambda x: (-x[1], x[0]))
    new_top_3 = [[prod, c] for prod, c in sorted_products[:3]]
    
    if new_top_3 != last_top_3:
        return ((product_counts, new_top_3), new_top_3)
    else:
        return ((product_counts, last_top_3), None)

top_3_stream = op.stateful_map('top_3', global_stream, update_global_top_3)
filtered = op.filter_map_value('filter_none', top_3_stream, lambda x: x)
no_key = op.key_rm('no_key', filtered)
formatted = op.map('format', no_key, lambda x: json.dumps(x))
op.output('out', formatted, StdOutSink())

run_main(flow)
