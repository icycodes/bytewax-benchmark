from datetime import datetime, timedelta, timezone
from bytewax.dataflow import Dataflow
import bytewax.operators as op
from bytewax.operators.windowing import SessionWindower, EventClock, collect_window
from bytewax.testing import TestingSource

flow = Dataflow("collect_window_example")

data = [
    ("u1", {"timestamp": "2023-01-01T10:00:00Z", "event_type": "login"}),
    ("u1", {"timestamp": "2023-01-01T10:00:02Z", "event_type": "view_item"}),
    ("u1", {"timestamp": "2023-01-01T10:00:08Z", "event_type": "add_to_cart"}),
    ("u2", {"timestamp": "2023-01-01T10:00:01Z", "event_type": "login"}),
    ("u2", {"timestamp": "2023-01-01T10:00:05Z", "event_type": "checkout"}),
]

up = op.input("input", flow, TestingSource(data))

def extract_timestamp(x):
    ts_str = x["timestamp"]
    if ts_str.endswith("Z"):
        ts_str = ts_str[:-1] + "+00:00"
    return datetime.fromisoformat(ts_str)

clock = EventClock(
    ts_getter=extract_timestamp,
    wait_for_system_duration=timedelta(seconds=0),
)

windower = SessionWindower(gap=timedelta(seconds=5))

windowed = collect_window("window", up, clock, windower)

def format_output(item):
    key, (window_id, values) = item
    return f"({key!r}, {values!r})"

formatted = op.map("format_output", windowed.down, format_output)

op.inspect("output", formatted)
EOF
